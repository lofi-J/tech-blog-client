## 화살표 함수 (Arrow Functions)

화살표 함수는 함수를 더 간결하게 작성할 수 있게 해주는 ES6의 핵심 기능입니다.

```javascript
// 기존 함수 선언
function add(a, b) {
  return a + b;
}

// 화살표 함수
const add = (a, b) => a + b;

// 매개변수가 하나일 때 괄호 생략 가능
const square = (x) => x * x;

// 매개변수가 없을 때
const getRandom = () => Math.random();

// 객체를 반환할 때
const createUser = (name, age) => ({ name, age });

// 여러 줄일 때
const processData = (data) => {
  const filtered = data.filter((item) => item.active);
  const sorted = filtered.sort((a, b) => a.id - b.id);
  return sorted.map((item) => ({ ...item, processed: true }));
};

// this 바인딩의 차이
class Timer {
  constructor() {
    this.seconds = 0;
  }

  start() {
    // 화살표 함수는 this를 렉시컬하게 바인딩
    setInterval(() => {
      this.seconds++;
      console.log(`${this.seconds}초 경과`);
    }, 1000);
  }
}
```

## 구조분해할당 (Destructuring)

구조분해할당은 배열이나 객체의 값을 개별 변수로 추출할 수 있게 해줍니다.

```javascript
// 배열 구조분해할당
const numbers = [1, 2, 3, 4, 5];
const [first, second, ...rest] = numbers;
console.log(first); // 1
console.log(second); // 2
console.log(rest); // [3, 4, 5]

// 객체 구조분해할당
const user = {
  name: "John Doe",
  age: 30,
  email: "john@example.com",
  address: {
    city: "Seoul",
    country: "Korea",
  },
};

const {
  name,
  age,
  address: { city },
} = user;
console.log(name); // 'John Doe'
console.log(age); // 30
console.log(city); // 'Seoul'

// 기본값 설정
const { role = "user", permissions = [] } = user;

// 함수 매개변수에서 구조분해할당
function createUser({ name, email, age = 18 }) {
  return { name, email, age };
}

const newUser = createUser({
  name: "Jane",
  email: "jane@example.com",
});

// 중첩된 구조분해할당
const config = {
  server: {
    host: "localhost",
    port: 3000,
    database: {
      name: "myapp",
      user: "admin",
    },
  },
};

const {
  server: {
    host,
    port,
    database: { name: dbName },
  },
} = config;
```

## 템플릿 리터럴 (Template Literals)

템플릿 리터럴은 문자열을 더 유연하고 읽기 쉽게 만들 수 있게 해줍니다.

```javascript
const name = "World";
const greeting = `Hello, ${name}!`;

// 여러 줄 문자열
const html = `
  <div class="container">
    <h1>${name}</h1>
    <p>Welcome to our website!</p>
  </div>
`;

// 표현식 사용
const price = 29.99;
const quantity = 3;
const total = `총 가격: ${price * quantity}원`;

// 함수 호출
function formatCurrency(amount) {
  return new Intl.NumberFormat("ko-KR", {
    style: "currency",
    currency: "KRW",
  }).format(amount);
}

const message = `주문 금액: ${formatCurrency(price * quantity)}`;

// 태그드 템플릿 리터럴
function highlight(strings, ...values) {
  let result = "";
  strings.forEach((string, i) => {
    result += string;
    if (values[i]) {
      result += `<span class="highlight">${values[i]}</span>`;
    }
  });
  return result;
}

const highlighted = highlight`Hello ${name}, welcome to ${"JavaScript"}!`;
```

## Promise와 async/await

Promise와 async/await는 비동기 프로그래밍을 더 쉽고 읽기 쉽게 만들어줍니다.

```javascript
// Promise 기본 사용법
function fetchUserData(userId) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      const user = {
        id: userId,
        name: "John Doe",
        email: "john@example.com",
      };

      if (userId > 0) {
        resolve(user);
      } else {
        reject(new Error("유효하지 않은 사용자 ID"));
      }
    }, 1000);
  });
}

// Promise 체이닝
fetchUserData(1)
  .then((user) => {
    console.log("사용자 정보:", user);
    return fetchUserPosts(user.id);
  })
  .then((posts) => {
    console.log("사용자 게시물:", posts);
  })
  .catch((error) => {
    console.error("에러:", error.message);
  });

// async/await 사용법
async function getUserInfo(userId) {
  try {
    const user = await fetchUserData(userId);
    const posts = await fetchUserPosts(user.id);

    return {
      user,
      posts,
      totalPosts: posts.length,
    };
  } catch (error) {
    console.error("사용자 정보 조회 실패:", error.message);
    throw error;
  }
}

// 병렬 실행
async function fetchAllData(userId) {
  const [user, posts, comments] = await Promise.all([
    fetchUserData(userId),
    fetchUserPosts(userId),
    fetchUserComments(userId),
  ]);

  return { user, posts, comments };
}

// Promise.allSettled 사용
async function fetchDataWithFallback(userIds) {
  const promises = userIds.map((id) => fetchUserData(id));
  const results = await Promise.allSettled(promises);

  const successful = results
    .filter((result) => result.status === "fulfilled")
    .map((result) => result.value);

  const failed = results
    .filter((result) => result.status === "rejected")
    .map((result) => result.reason);

  return { successful, failed };
}
```

## 모듈 시스템 (ES6 Modules)

ES6 모듈 시스템은 코드를 더 모듈화하고 재사용 가능하게 만들어줍니다.

```javascript
// utils.js
export const formatDate = (date) => {
  return new Intl.DateTimeFormat("ko-KR").format(date);
};

export const validateEmail = (email) => {
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email);
};

export default class Logger {
  constructor(prefix = "") {
    this.prefix = prefix;
  }

  log(message) {
    console.log(`${this.prefix}: ${message}`);
  }

  error(message) {
    console.error(`${this.prefix}: ERROR - ${message}`);
  }
}

// userService.js
import Logger, { validateEmail } from "./utils.js";

export class UserService {
  constructor() {
    this.logger = new Logger("UserService");
  }

  async createUser(userData) {
    if (!validateEmail(userData.email)) {
      throw new Error("유효하지 않은 이메일 주소");
    }

    this.logger.log(`새 사용자 생성: ${userData.email}`);
    // 사용자 생성 로직...
  }
}

// main.js
import { UserService } from "./userService.js";
import { formatDate } from "./utils.js";

const userService = new UserService();
console.log(`현재 시간: ${formatDate(new Date())}`);
```
